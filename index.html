  Example Javascript Code

var mihenkdata = {
   "xdata": [1, 2, 3, 4, 5],
   "ydata": [4, 5, 6, 7, 8],
   "opt": {
     "x_labels": [],
     "y_labels": [],
     "zoom": true,
     "xlab": "x label",
     "ylab": "y label",
     "color": [],
     "div_id": "graph"
   }
};

mihenk.barchart(mihenkdata);
